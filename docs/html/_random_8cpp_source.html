72UINT CRandom::randn(){
73 UINT s = m_uState[3];
74
75 s ^= s << 11;
76 s ^= s >> 8;
77
78 m_uState[3] = m_uState[2];
79 m_uState[2] = m_uState[1];
80 m_uState[1] = m_uState[0];
81
82 s ^= m_uState[0];
83 s ^= m_uState[0] >> 19;
84
85 m_uState[0] = s;
86
87 return s;
88} //randn
