117 void CMain::Draw(){
118  const UINT nTileWidth = m_pTile[0]->GetWidth();
119  const UINT nTileHeight = m_pTile[0]->GetHeight();
120 
121  if(m_pBitmap == nullptr){
122  const int w = int(nTileWidth*m_pWangTiler->GetWidth());
123  const int h = int(nTileHeight*m_pWangTiler->GetHeight());
124  m_pBitmap = new Gdiplus::Bitmap(w, h);
125  } //if
126 
127  const size_t nGridWidth = m_pWangTiler->GetWidth();
128  const size_t nGridHeight = m_pWangTiler->GetHeight();
129 
130  Gdiplus::Rect r;
131  r.Width = nTileWidth;
132  r.Height = nTileHeight;
133 
134  Gdiplus::Graphics graphics(m_pBitmap);
135  graphics.SetSmoothingMode(Gdiplus::SmoothingModeHighQuality);
136 
137  for(size_t i=0; i<nGridHeight; i++){
138  for(size_t j=0; j<nGridWidth; j++){
139  const size_t index = (*m_pWangTiler)(i, j);
140  graphics.DrawImage(m_pTile[index], r);
141  r.X += nTileWidth; //next column
142  } //for
143 
144  r.Y += nTileHeight; //next row
145  r.X = 0; //first column
146  } //for
147 } //Draw

194 HRESULT CMain::LoadTileSet(const UINT idm, const UINT n){
195  UINT i = 0;
196  bool error = false;
197  std::wstring filename;
198  Gdiplus::Bitmap** temp = new Gdiplus::Bitmap*[n];
199 
200  if(m_nNumTiles != n){
201  for(UINT i=0; i<m_nNumTiles; i++)
202  delete m_pTile[i];
203  delete [] m_pTile;
204 
205  m_pTile = new Gdiplus::Bitmap*[n];
206  for(UINT i=0; i<n; i++)
207  m_pTile[i] = nullptr;
208 
209  m_nNumTiles = n;
210  } //if
211 
212  for(i=0; i<n && !error; i++){ //for each tile
213  filename = L"tiles\\"; //file name
214 
215  switch(idm){
216  case IDM_TILESET_DEFAULT: filename += L"default"; break;
217  case IDM_TILESET_FLOWER: filename += L"flowers"; break;
218  case IDM_TILESET_DIRT: filename += L"dirt"; break;
219  } //switch
220 
221  filename += std::wstring(L"\\") + std::to_wstring(i) + L".png";
222 
223  temp[i] = Gdiplus::Bitmap::FromFile(filename.c_str()); //load tile
224  error = temp[i]->GetLastStatus() != Gdiplus::Ok;
225  } //for
226 
227  //error handling
228 
229  if(error){ //fail
230  std::wstring s = L"Error loading file " + filename;
231  MessageBoxW(m_hWnd, s.c_str(), L"Error", MB_ICONERROR | MB_OK);
232 
233  for(UINT j=0; j<i; j++)
234  delete temp[j];
235  } //if
236 
237  else{ //success
238  for(UINT j=0; j<n; j++){ //copy tile pointers to m_pTile
239  delete m_pTile[j];
240  m_pTile[j] = temp[j];
241  } //for
242 
243  //unset menu checkmarks then check the one we want
244  CheckMenuItem(m_hTilesetMenu, IDM_TILESET_DEFAULT, MF_UNCHECKED);
245  CheckMenuItem(m_hTilesetMenu, IDM_TILESET_FLOWER, MF_UNCHECKED);
246  CheckMenuItem(m_hTilesetMenu, IDM_TILESET_DIRT, MF_UNCHECKED);
247 
248  CheckMenuItem(m_hTilesetMenu, idm, MF_CHECKED);
249  } //else
250 
251  delete [] temp;
252  return error? E_FAIL: S_OK;
253 } //LoadTileSet
