64UINT CWangTiler::Match(UINT x, UINT y){
65 std::uniform_int_distribution<UINT> d(0, 1);
66 return (y&4) ^ (y&1)<<2 | (x&2) ^ (x&1)<<1 | d(m_stdRandom);
67} //Match

72void CWangTiler::Generate(){
73 std::uniform_int_distribution<UINT> d(0, 7);
74
75 m_nTile[0][0] = d(m_stdRandom);
76
77 for(size_t j=1; j<m_nWidth; j++)
78 m_nTile[0][j] = Match(m_nTile[0][j - 1], d(m_stdRandom));
79
80 for(size_t i=1; i<m_nHeight; i++){
81 m_nTile[i][0] = Match(d(m_stdRandom), m_nTile[i - 1][0]);
82
83 for(size_t j=1; j<m_nWidth; j++)
84 m_nTile[i][j] = Match(m_nTile[i][j - 1], m_nTile[i - 1][j]);
85 } //for
86} //Generate
